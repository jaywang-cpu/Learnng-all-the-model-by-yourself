```
               pickle.dump（le,f)

(完整简洁版代码）
colors = ['红色', '蓝色', '绿色', '红色', '蓝色']
le_colors = LabelEncoder()
colors_encoded = le_colors.fit_transform(colors)
with open('color_encoder.pkl', 'wb') as f:
    pickle.dump(le_colors, f)
with open('color_encoder.pkl', 'rb') as f:
    encoder = pickle.load(f)
new_colors = ['红色', '绿色']
result = encoder.transform(new_colors)
print(f"新数据 {new_colors} 编码为: {result}")
```
